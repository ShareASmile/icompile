Make The Previous Post A Mass APK Installer That Does Not Uses ADB Install-Multi Syntax

Posted at Sat, 23 Jul 2016 18:14:55 GMT 

@echo off



:loop

      ::-------------------------- has argument ?

      if ["%~1"]==[""] (

        echo done.

        goto end

      )

      ::-------------------------- argument exist ?

      if not exist %~s1 (

        echo error "%~1" does not exist in file-system.

      ) else (

        echo "%~1" exist

        if exist %~s1\\NUL (

          echo "%~1" is a directory

        ) else (

          echo "%~1" is a file! - time to install:

          call adb install %~s1

        )

      )

      ::--------------------------

      shift

      goto loop

      

      

:end



pause
