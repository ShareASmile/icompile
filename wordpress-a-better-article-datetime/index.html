Note II:
You can force the "emoji" symbol to render either as text or picture by adding a Unicode-variation-selector (mostly effective only on mobiles/iOS desktops)
to keep the emoji "as text" use &#x1F4C5;&#xFE0E; or &#128197;&#65038; (known as Unicode's VARIATION SELECTOR-15).
to keep the emoji "as picture" use &#x1F4C5;&#xFE0F; or &#128197;&#65039; (known as Unicode's VARIATION SELECTOR-16).

Again.. this, most likely, won't change anything in Chrome (*nix/Win Desktops), only on Android devices and iOS desktops and mobile devices.

var xhr = new XMLHttpRequest();

xhr.open("HEAD", "https://eladkarako.github.io/trackers/trackers.txt", true);

xhr.onreadystatechange = function(){

  "use strict";



  if(XMLHttpRequest.DONE !== xhr.readyState)  return;



  document.body.innerHTML = 

  (function(DAYS, MONTHS, ORDINAL, date

          , day, day_of_week, month, year, hour, minute, ampm, DATE_c, DATE_lFj, DATE_S, DATE_YhiA, DATE_e){



    day         = date.getUTCDate();

    day_of_week = date.getUTCDay();

    month       = date.getUTCMonth();

    year        = date.getUTCFullYear();

    hour        = date.getUTCHours();

    minute      = date.getUTCMinutes();

    ampm        = (0 === hour && 1 <= minute || 0 < hour && 12 > hour) ? "AM" : "PM";



    

    DATE_c      = date.toISOString().replace(/\\..*$/g,"") + "+00:00";                   /* "2016-05-06T20:31:18+00:00"  Note: HTML5 datetime uses ISO 8601 date format*/

    DATE_lFj    = DAYS[day_of_week] + ", " + MONTHS[month] + " " + String(day);         /* "Friday, May 6"                       */

    DATE_S      = ORDINAL[day];                                                         /* "th"                                  */

    DATE_YhiA   = ", " + String(year) + ".  " + String(hour) + ":" + minute + ampm;     /* ", 2016.  8:31PM"                     */

    DATE_e      = "UTC";                                                                /* "UTC"  Note: we alway use UTC here... */



    date        = '📅︎ <time datetime="##DATE_c##" pubdate>##DATE_lFj##<sup>##DATE_S##</sup>##DATE_YhiA##</time>';

    date        = date.replace("##DATE_c##",       DATE_c)

                      .replace("##DATE_lFj##",   DATE_lFj)

                      .replace("##DATE_S##",       DATE_S)

                      .replace("##DATE_YhiA##", DATE_YhiA)

                      ;

    return date;

  }(
